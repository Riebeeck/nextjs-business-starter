## Technical Implementation

### Real-Time Updates with Firebase

```typescript
import { 
  collection, 
  onSnapshot, 
  query, 
  where 
} from 'firebase/firestore';

function useTasksRealtime(workspaceId: string) {
  const [tasks, setTasks] = useState<Task[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const q = query(
      collection(db, 'tasks'),
      where('workspaceId', '==', workspaceId)
    );

    const unsubscribe = onSnapshot(q, (snapshot) => {
      const taskData = snapshot.docs.map(doc => ({
        id: doc.id,
        ...doc.data()
      })) as Task[];
      
      setTasks(taskData);
      setLoading(false);
    });

    return () => unsubscribe();
  }, [workspaceId]);

  return { tasks, loading };
}
```

### Drag-and-Drop Implementation

```typescript
import { DndContext, closestCenter, DragEndEvent } from '@dnd-kit/core';
import { SortableContext, verticalListSortingStrategy } from '@dnd-kit/sortable';

export default function TaskBoard() {
  const { tasks } = useTasksRealtime(workspaceId);

  const handleDragEnd = async (event: DragEndEvent) => {
    const { active, over } = event;
    
    if (!over || active.id === over.id) return;

    await updateTaskColumn(active.id, over.id);
  };

  return (
    <DndContext 
      collisionDetection={closestCenter}
      onDragEnd={handleDragEnd}
    >
      <div className="flex gap-4">
        {columns.map(column => (
          <SortableContext
            key={column.id}
            items={tasks.filter(t => t.columnId === column.id)}
            strategy={verticalListSortingStrategy}
          >
            <TaskColumn column={column} tasks={tasks} />
          </SortableContext>
        ))}
      </div>
    </DndContext>
  );
}
```

### Optimistic UI Updates

```typescript
async function updateTask(taskId: string, updates: Partial<Task>) {
  // Update UI immediately
  setTasks(prev => 
    prev.map(task => 
      task.id === taskId 
        ? { ...task, ...updates } 
        : task
    )
  );

  try {
    // Update in database
    await updateDoc(doc(db, 'tasks', taskId), updates);
  } catch (error) {
```

## Architecture Decisions

### Why Firebase?

**Pros:**
- Real-time updates out of the box
- No backend server needed
- Authentication included
- Automatic scaling
- Free tier for development

**Cons:**
- Lock-in to Google ecosystem
- Complex queries can be expensive
- Limited query capabilities compared to SQL

**Decision:** Firebase was perfect for MVP. Real-time updates were essential and Firebase made them trivial to implement.

### State Management

Used a combination of:
- **React Context** - User authentication and workspace selection
- **React Query** - Server state and caching
- **Local State** - UI state (modals, forms, etc.)

Avoided Redux because the app didn't need global state management complexity.

## Challenges & Solutions

### Challenge 1: Real-Time Performance

**Problem:** Real-time updates caused excessive re-renders.

**Solution:**
- Implemented React.memo for task components
- Used useMemo for expensive calculations
- Batched Firebase updates
- Result: 60fps performance with 1000+ tasks

### Challenge 2: Optimistic Updates

**Problem:** Network lag made the UI feel slow.

**Solution:**
- Implemented optimistic updates
- Added loading states for network requests
- Showed undo option for important actions
- Result: Instant-feeling UI

### Challenge 3: Offline Support

**Problem:** App broke without internet connection.

**Solution:**
- Enabled Firebase offline persistence
- Added network status indicator
- Queued actions while offline
- Result: Works offline, syncs when back online

## Security & Permissions

### Role-Based Access Control

```typescript
interface Workspace {
  id: string;
  name: string;
  members: {
    [userId: string]: 'owner' | 'admin' | 'member';
  };
}

function canEditTask(user: User, task: Task, workspace: Workspace): boolean {
  const role = workspace.members[user.id];
  
  // Owners and admins can edit any task
  if (role === 'owner' || role === 'admin') {
    return true;
  }
  
  // Members can only edit their assigned tasks
  return task.assignedTo === user.id;
}
```

### Firestore Security Rules

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /tasks/{taskId} {
      allow read: if request.auth != null && 
        exists(/databases/$(database)/documents/workspaces/$(resource.data.workspaceId)/members/$(request.auth.uid));
      
      allow create: if request.auth != null;
      
      allow update, delete: if request.auth != null && 
        (resource.data.assignedTo == request.auth.uid || 
         get(/databases/$(database)/documents/workspaces/$(resource.data.workspaceId)).data.members[request.auth.uid] in ['owner', 'admin']);
    }
  }
}
```
